beta = data.frame(matrix(rnorm(1000),200,5))
names(beta) = c("beta0","beta1", "beta2", "beta3", "beta4")


layout(matrix(1:3,1, 3))
boxplot(beta, col=8, main = "boxplot")
vioplot(beta[,1], beta[,2], beta[,3], beta[,4], beta[,5], col = 8, names=names(beta))
title("vioplot")
violinplot(beta, col=8)
